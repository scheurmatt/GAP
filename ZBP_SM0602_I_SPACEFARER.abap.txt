  PROTECTED SECTION.

    METHODS cleanup_finalize REDEFINITION.

ENDCLASS.

CLASS lsc_zsm0602_i_spacefarer IMPLEMENTATION.

  METHOD cleanup_finalize.
  ENDMETHOD.

ENDCLASS.

CLASS lhc_ZSM0602_I_SPACEFARER DEFINITION INHERITING FROM cl_abap_behavior_handler.
  PRIVATE SECTION.

    METHODS get_instance_features FOR INSTANCE FEATURES
      IMPORTING keys REQUEST requested_features FOR zsm0602_i_spacefarer RESULT result.

    METHODS get_instance_authorizations FOR INSTANCE AUTHORIZATION
      IMPORTING keys REQUEST requested_authorizations FOR zsm0602_i_spacefarer RESULT result.

    METHODS validate_reputation FOR VALIDATE ON SAVE
      IMPORTING keys FOR zsm0602_i_spacefarer~validate_reputation.

ENDCLASS.

CLASS lhc_ZSM0602_I_SPACEFARER IMPLEMENTATION.

  METHOD get_instance_features.
  ENDMETHOD.

  METHOD get_instance_authorizations.
  ENDMETHOD.

  METHOD validate_reputation.
  READ ENTITIES OF zsm0602_i_spacefarer IN LOCAL MODE
    ENTITY zsm0602_i_spacefarer
    FIELDS ( reputation navigation_skill )
    WITH CORRESPONDING #( keys )
    RESULT DATA(lt_spacefarer).

  LOOP AT lt_spacefarer INTO DATA(ls_sf).

    DATA(lv_total) = ls_sf-reputation + ls_sf-navigation_skill.

    IF lv_total > 99.

      APPEND VALUE #(
        %tky = ls_sf-%tky
      ) TO failed-zsm0602_i_spacefarer.

      APPEND VALUE #(
        %tky = ls_sf-%tky
        %msg = new_message(
          id       = 'ZSM0602_MSG'
          number   = '001'
          severity = if_abap_behv_message=>severity-error
          v1       = |Reputation + Navigation Skill must be <= 99|
        )
      ) TO reported-zsm0602_i_spacefarer.

    ENDIF.

  ENDLOOP.
  ENDMETHOD.

ENDCLASS.
